Expand description

The trait that enables something to be have a style.

Examples

use ratatui::{
    style::{Color, Modifier, Style, Styled, Stylize},
    text::Span,
};

assert_eq!(
   "hello".red().on_blue().bold(),
    Span::styled("hello", Style::default().fg(Color::Red).bg(Color::Blue).add_modifier(Modifier::BOLD))
)
